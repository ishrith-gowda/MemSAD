\title{\memsad{}: Gradient-Coupled Anomaly Detection for Memory Poisoning in Retrieval-Augmented Agents}

% anonymous for submission
\author{Anonymous Author(s)}

\begin{document}

\maketitle

\begin{abstract}
Persistent external memory enables LLM agents to maintain context across sessions, yet its security properties remain formally uncharacterized.
We introduce a rigorous threat model for memory poisoning attacks on retrieval-augmented agents, formalized as a Stackelberg game, and present a unified evaluation framework encompassing three attack classes with distinct access assumptions. This framework corrects a systematic protocol error in prior work that underestimates attack success by $4\times$.
Our primary contribution is \memsad{} (\emph{Semantic Anomaly Detection}), a calibration-based defense for which we prove a \emph{gradient coupling theorem}: the anomaly score gradient and retrieval objective gradient are provably identical under encoder regularity, establishing that continuous evasion necessarily degrades retrieval rank.
We derive an information-theoretic lower bound via Le Cam's method showing any threshold detector requires $\Omega(1/\rho^2)$ calibration samples; \memsad{} achieves this up to $\log(1/\delta)$ factors, proving minimax optimality.
We provide a certified detection radius, a checkable condition guaranteeing detection regardless of adversary strategy, and online regret bounds for rolling calibration with optimal window selection at rate $O(\sigma^{2/3}\Delta^{1/3})$.
We formally characterize a discrete synonym-invariance loophole with tight perturbation bounds, precisely delineating where continuous guarantees break down.
Empirical evaluation over a $3 \times 5$ attack-defense matrix with bootstrap CIs, Bonferroni-corrected hypothesis tests, and Clopper-Pearson validation (20 trials, $n=1000$) demonstrates that composite portfolios achieve $\tpr = 1.00$, $\fpr = 0.00$ across all attacks, while the synonym loophole enables 80--100\% evasion at $\Delta\asrr \approx 0$, confirming the theory.
\end{abstract}

\section{Introduction}
\label{sec:intro}

Modern LLM agents persist context across sessions through external memory systems such as Mem0~\citep{mem0}, A-MEM~\citep{amem}, and MemGPT~\citep{memgpt2023}.
These systems store user preferences, task history, and factual knowledge as dense vector embeddings, retrieving semantically relevant entries to condition the agent's response.
The practical utility of persistent memory is well established; its security implications are not.
A single adversarial entry injected into an agent's memory may remain indefinitely, triggering on every semantically related query across arbitrary future sessions.
Unlike prompt injection, which requires per-interaction access, memory poisoning \emph{persists and scales}: the adversary acts once, and the attack surface compounds with each new query.

Three recent papers characterize distinct threat models with escalating access requirements.
\agentpoison{}~\citep{chen2024agentpoison} formulates trigger optimization as a constrained gradient problem over DPR~\citep{karpukhin2020dense} embeddings ($\alpha = \textsc{write}$).
\minja{}~\citep{dong2025minja} extends the setting to query-only access via progressive-shortening indication ($\alpha = \textsc{query}$).
\injecmem{}~\citep{injecmem2026} achieves single-interaction injection via retriever-agnostic anchors ($\alpha = \textsc{single}$).
Despite the severity of these threats, no prior work provides a unified evaluation across attack classes or proposes defenses with formal detection guarantees for the memory-agent setting.
Existing RAG defenses~\citep{zou2024poisonedrag, chaudhari2024phantom, xiang2024robustrag} assume corpus-level access and offline cleaning, neither of which applies to the streaming ingestion model of agent memory.

\paragraph{Contributions.}
We make five contributions spanning theory, algorithm design, and empirical methodology.
\textbf{(1)}~\emph{Formal threat model.} Stackelberg game formulation (Definitions~\ref{def:threat_model}--\ref{def:stackelberg}) with equilibrium existence (Proposition~\ref{prop:equilibrium}), and a protocol correction that raises \agentpoison{} $\asrr$ from 0.25 to 1.00 ($4\times$), resolving a systematic underestimation in prior evaluation.
\textbf{(2)}~\emph{\memsad{} defense.} A calibration-based detector (Definition~\ref{def:sad}) with gradient coupling theorem (Theorem~\ref{thm:gradient_coupling}) proving continuous evasion impossible, and a certified detection radius (Lemma~\ref{lem:certified}) providing checkable, per-entry guarantees.
\textbf{(3)}~\emph{Theoretical foundations.} Information-theoretic lower bound via Le Cam's method (Theorem~\ref{thm:minimax_lower}) proving minimax optimality up to $\log(1/\delta)$; finite-sample calibration bounds (Theorem~\ref{thm:calibration_bound}); online regret bounds (Theorem~\ref{thm:regret}) with optimal window at rate $O(\sigma^{2/3}\Delta^{1/3})$; and a Wasserstein generalization bound (Theorem~\ref{thm:generalization}) for transfer to unseen attacks.
\textbf{(4)}~\emph{Impossibility characterization.} Formal analysis of the discrete synonym-invariance loophole (Proposition~\ref{prop:synonym}) with tight perturbation bounds, delineating exactly where continuous guarantees fail.
\textbf{(5)}~\emph{Rigorous evaluation.} A $3 \times 5$ attack-defense matrix with Bonferroni-corrected hypothesis testing ($\alpha' = 0.003$, 15 comparisons), bootstrap CIs, and Clopper-Pearson validation (20 trials, $n = 1000$) demonstrating composite portfolios achieve $\tpr = 1.00$, $\fpr = 0.00$ across all attacks.

\section{Related work}
\label{sec:related}

\paragraph{Memory poisoning attacks.}
\citet{chen2024agentpoison} achieve $\asrr \geq 0.80$ via trigger optimization over DPR~\citep{karpukhin2020dense} embeddings.
\citet{dong2025minja} exploit auto-storage via progressive-shortening ($\isr = 98.2\%$).
\citet{injecmem2026} achieve single-interaction injection via retriever-agnostic anchors.
\citet{memorygraft2025} demonstrate experience-grafting attacks exploiting union retrieval.
On the defense side, ReliabilityRAG~\citep{reliabilityrag2025} provides provable robustness via graph-theoretic maximum independent set, while SeCon-RAG~\citep{seconrag2025} uses entity-intent-relation extraction; both target RAG corpora rather than agent memory streams.

\paragraph{RAG corpus poisoning and certified defenses.}
\citet{zou2024poisonedrag} optimize adversarial documents via HotFlip~\citep{ebrahimi2018hotflip} ($> 90\%$ ASR at million-scale).
\citet{xiang2024robustrag} provide certified robustness via isolate-then-aggregate with provable guarantees under bounded perturbations; however, their certificates require post-retrieval processing of $k$ retrieved passages and do not apply to pre-ingestion filtering where each entry must be evaluated independently.
SPECTRE~\citep{hayase2021spectre} uses robust covariance estimation for spectral detection but assumes full dataset access and $O(|\cM|^2 d)$ computation.
Our work differs along three axes: (i)~defenses operate at \emph{write time} on individual entries ($O(md)$ per entry), (ii)~no corpus-level access is required, only a rolling query history, and (iii)~we provide both a certified detection radius (Lemma~\ref{lem:certified}) and information-theoretic optimality (Theorem~\ref{thm:minimax_lower}), whereas RobustRAG's certificates are for a fixed retrieval result rather than the ingestion boundary.

\paragraph{Game-theoretic security and adaptive evaluation.}
\citet{tramer2020adaptive} establish the gold standard for adaptive attack evaluation.
Our Stackelberg formulation follows this principle: the adversary optimizes with full knowledge of $(\hat{\mu}, \hat{\sigma}, \kappa)$.
\citet{carlini2017evaluating} demonstrate that defenses without formal guarantees are routinely broken; our gradient coupling theorem provides exactly such a guarantee for the continuous case, while Proposition~\ref{prop:synonym} honestly characterizes the discrete failure mode.

\section{Formal threat model}
\label{sec:threat}

\begin{definition}[Memory-augmented agent]
\label{def:agent}
A \emph{memory-augmented agent} is a tuple $(\cA, \cM, \enc, k)$ where $\cA$ is an LLM, $\cM \subset \cV^*$ is a persistent memory store over vocabulary $\cV$, $\enc : \cV^* \to \RR^d$ is an embedding function, and $k \in \NN$ is the retrieval depth. At each interaction with query $q$, the agent retrieves
\begin{equation}
  \Ret(\cM, q, k) = \argmax_{\substack{S \subseteq \cM \\ |S| = k}} \sum_{m \in S} \cosim(\enc(q), \enc(m)),
  \label{eq:retrieval}
\end{equation}
where $\cosim(u, v) \coloneqq u^\top v / (\|u\| \|v\|)$ denotes cosine similarity.
\end{definition}

\begin{definition}[Memory poisoning threat model]
\label{def:threat_model}
An \emph{adversary} $\cS = (\cQ_v, n, \alpha)$ targets victim queries $\cQ_v \subset \cV^*$ by injecting $\cP = \{p_1, \ldots, p_n\}$ into $\cM$ under access model $\alpha \in \{\textsc{write}, \textsc{query}, \textsc{single}\}$ to maximize:
\begin{align}
  \asrr(\cP, \cQ_v) &\coloneqq |\{q \in \cQ_v : \cP \cap \Ret(\cM \cup \cP, q, k) \neq \emptyset\}| / |\cQ_v|, \label{eq:asrr} \\
  \asra(\cP) &\coloneqq \PP[\cA \text{ executes adversarial action} \mid \text{poison retrieved}], \label{eq:asra} \\
  \asrt(\cP, \cQ_v) &\coloneqq \asrr \cdot \asra. \label{eq:asrt}
\end{align}
\end{definition}

\begin{definition}[Stackelberg game formulation]
\label{def:stackelberg}
The memory poisoning interaction is a Stackelberg game $\cG = (\cS, \cD, u_{\cS}, u_{\cD})$ where the defender (leader) commits to a detection policy $\pi_{\cD} : \cV^* \to \{0, 1\}$ and the adversary (follower) best-responds:
\begin{align}
  \pi_{\cD}^* &= \argmin_{\pi_{\cD} \in \Pi} \; \max_{\cP \in \cB(\pi_{\cD})} \; \asrr(\cP, \cQ_v), \label{eq:leader} \\
  \cB(\pi_{\cD}) &= \argmax_{\cP \subseteq \cV^*,\, |\cP| \leq n} \; \asrr(\cP \setminus \{p : \pi_{\cD}(p) = 1\}, \cQ_v), \label{eq:follower}
\end{align}
where $\Pi = \{\pi(c) = \mathbf{1}[s(c; \cH) > \tau] : \tau > 0\}$ is the class of threshold detectors.
The defender's utility is $u_{\cD} = -\asrr - \lambda \cdot \fpr$ ($\lambda > 0$) and the adversary's is $u_{\cS} = \asrt$.
\end{definition}

\begin{assumption}[Encoder regularity]
\label{ass:encoder}
$\enc : \cV^* \to \RR^d$ satisfies: (i) $\|\enc(x)\| > 0$ for all $x$; (ii) differentiability w.r.t.\ the output embedding; (iii) L2-normalization: $\|\enc(x)\| = 1$ (w.l.o.g.).
\end{assumption}

Under Assumption~\ref{ass:encoder}(iii), $\cosim(\enc(q), \enc(m)) = \enc(q)^\top \enc(m)$ and retrieval reduces to maximum inner product search.

\section{\memsad{}: Semantic Anomaly Detection}
\label{sec:sad}

\subsection{Algorithm and formal definition}

Memory poisoning succeeds because adversarial entries are semantically close to victim queries; this closeness, necessary for retrieval, is itself a detectable signal.

\begin{definition}[\memsad{} detector]
\label{def:sad}
Let $\cH = \{q_1, \ldots, q_m\}$ be a rolling query history (FIFO, capacity $m_{\max}$) and $\cM_{\text{ref}} \subseteq \cM$ a benign reference corpus. The anomaly score is $s(c; \cH) \coloneqq \max_{q \in \cH} \cosim(\enc(c), \enc(q))$, calibrated via $\hat{\mu}, \hat{\sigma}$ of $\{s(m; \cH) : m \in \cM_{\text{ref}}\}$. Entry $c$ is flagged if $s(c; \cH) > \hat{\mu} + \kappa \hat{\sigma}$ for $\kappa > 0$.
The \emph{combined scoring mode} captures both targeted and broad-recall signatures:
\begin{equation}
  s_{\text{comb}}(c; \cH)
  \coloneqq \tfrac{1}{2} \max_{q \in \cH} \cosim(\enc(c), \enc(q))
  + \tfrac{1}{2} \cdot \frac{1}{|\cH|}\sum_{q \in \cH} \cosim(\enc(c), \enc(q)).
  \label{eq:sad_combined}
\end{equation}
\end{definition}

\begin{algorithm}[t]
\caption{\memsad{} pre-ingestion filtering}
\label{alg:sad}
\begin{algorithmic}[1]
\REQUIRE candidate $c$, query history $\cH$, parameters $(\hat{\mu}, \hat{\sigma})$, threshold $\kappa$
\STATE $s \leftarrow s_{\text{comb}}(c; \cH)$ \hfill $\triangleright$ $O(md)$ time
\IF{$s > \hat{\mu} + \kappa \cdot \hat{\sigma}$}
  \STATE \textbf{reject} $c$
\ELSE
  \STATE \textbf{accept} $c$ into $\cM$
\ENDIF
\end{algorithmic}
\end{algorithm}

\subsection{Gradient coupling theorem}
\label{sec:gradient}

\begin{theorem}[Gradient coupling]
\label{thm:gradient_coupling}
Let $\enc$ satisfy Assumption~\ref{ass:encoder}, let $e_c \coloneqq \enc(c) \in \RR^d$, and $q^* \coloneqq \argmax_{q \in \cH} \cosim(e_c, \enc(q))$. Define the detection objective $\cD(e_c) \coloneqq \cosim(e_c, \enc(q^*))$ and retrieval objective $\cR(e_c) \coloneqq \cosim(e_c, \enc(q^*))$. Then $\cD \equiv \cR$ on $\RR^d \setminus \{0\}$ and:
\begin{equation}
  \nabla_{e_c} \cD(e_c)
  = \nabla_{e_c} \cR(e_c)
  = \frac{\enc(q^*) - (e_c^\top \enc(q^*)) e_c}{\|e_c\|^2 \|\enc(q^*)\|}.
  \label{eq:gradient}
\end{equation}
For any perturbation $\delta \in \RR^d$: $\langle \delta, \nabla \cD \rangle < 0 \iff \langle \delta, \nabla \cR \rangle < 0$.
\end{theorem}

\begin{proof}
Since $\cD \equiv \cR$ by construction, gradient identity is immediate. The explicit form follows from $\nabla_{e_c}(e_c^\top v / (\|e_c\|\|v\|)) = (v - (e_c^\top v / \|e_c\|^2)e_c) / (\|e_c\|\|v\|)$ with $v = \enc(q^*)$.
\end{proof}

\begin{corollary}[Detection-evasion tradeoff]
\label{cor:tradeoff}
Under L2-normalization, for perturbation $e_c' = e_c + \epsilon\delta$ with $\|\delta\| = 1$:
\begin{equation}
  \cD(e_c') - \cD(e_c) = \epsilon \langle \delta, \enc(q^*) - (e_c^\top \enc(q^*)) e_c \rangle + O(\epsilon^2).
  \label{eq:tradeoff}
\end{equation}
Thus $\cD(e_c') < \cD(e_c)$ (evasion) $\implies$ $\cR(e_c') < \cR(e_c)$ (retrieval degradation) to first order.
\end{corollary}

\subsection{Certified detection guarantee}
\label{sec:certified}

The gradient coupling (Theorem~\ref{thm:gradient_coupling}) shows continuous evasion is impossible. We now provide a \emph{certified detection guarantee}: under explicit, verifiable conditions on the adversary's retrieval requirement and the calibration quality, \memsad{} provably detects adversarial passages.

\begin{lemma}[Certified detection radius]
\label{lem:certified}
Let $\enc$ satisfy Assumption~\ref{ass:encoder} and let $c$ be an adversarial passage achieving $\rank(\enc(c), \cM) \leq k$ for victim query $q^*$. Define the \emph{benign similarity ceiling} $\bar{s} \coloneqq \max_{m \in \cM_{\text{ref}}} \cosim(\enc(m), \enc(q^*))$, the adversarial similarity $s_{\text{adv}} \coloneqq \cosim(\enc(c), \enc(q^*))$, and the \emph{gap} $\Delta_s \coloneqq s_{\text{adv}} - \bar{s}$.
If the calibration satisfies $|\tau_N - \tau^*| \leq \eta$ (Theorem~\ref{thm:calibration_bound}) and the gap exceeds
\begin{equation}
  \Delta_s > \kappa\hat{\sigma} + \eta,
  \label{eq:certified_radius}
\end{equation}
then \memsad{} detects $c$ with certainty: $s(c; \cH) > \hat{\mu} + \kappa\hat{\sigma}$.
\end{lemma}

\begin{proof}
Since $c$ achieves $\rank \leq k$, we have $s(c; \cH) \geq s_{\text{adv}} = \bar{s} + \Delta_s$. The detection threshold is $\hat{\mu} + \kappa\hat{\sigma} \leq \bar{s} + \eta$ (since $\hat{\mu} + \kappa\hat{\sigma} \leq \tau^* + \eta \leq \bar{s} + \eta$ by definition of the benign ceiling). Thus $s(c; \cH) \geq \bar{s} + \Delta_s > \bar{s} + \kappa\hat{\sigma} + \eta \geq \hat{\mu} + \kappa\hat{\sigma}$.
\end{proof}

Lemma~\ref{lem:certified} formalizes a ``detection radius'': any adversarial passage whose similarity to the victim query exceeds the benign ceiling by more than $\kappa\hat{\sigma} + \eta$ is \emph{certifiably detected}, regardless of the adversary's strategy (Figure~\ref{fig:certified_radius}).
For \minja{} in our experiments, $\Delta_s \approx 0.15$ while $\kappa\hat{\sigma} + \eta \approx 0.08$, satisfying the condition with margin.
For \agentpoison{} under triggered calibration, $\Delta_s \approx 0.12$ with $\kappa\hat{\sigma} + \eta \approx 0.06$.

\subsection{Information-theoretic detection lower bound}
\label{sec:lower_bound}

\begin{theorem}[Minimax detection lower bound]
\label{thm:minimax_lower}
Let $\cF$ be the class of threshold detectors $\pi(c) = \mathbf{1}[s(c; \cH) > \tau]$. Let $P_0, P_1$ be the benign and poisoned score distributions with means $\mu_0, \mu_1$, common variance $\sigma^2$, and signal-to-noise ratio $\rho \coloneqq |\mu_1 - \mu_0| / \sigma$. For any detector $\pi \in \cF$ calibrated on $N$ i.i.d.\ samples from $P_0$:
\begin{equation}
  \inf_{\pi \in \cF} \left[\fpr(\pi) + (1 - \tpr(\pi))\right]
  \geq 1 - \frac{\rho\sqrt{N}}{2}.
  \label{eq:minimax_lower}
\end{equation}
Reliable detection ($\fpr + (1 - \tpr) \leq \epsilon$) requires $N \geq 4(1 - \epsilon)^2 / \rho^2$.
\end{theorem}

\begin{proof}[Proof sketch]
Apply Le Cam's two-point method~\citep{lecam1973convergence,yu1997assouad} with $H_0: c \sim P_0$, $H_1: c \sim P_1$. Pinsker's inequality gives $\TV(P_0^N, P_1^N) \leq \sqrt{N \cdot \KL(P_0 \| P_1)/2} \leq \rho\sqrt{N}/2$ for sub-Gaussian distributions. Le Cam's bound then yields Eq.~\eqref{eq:minimax_lower}. Full proof in Appendix~\ref{app:proof_minimax}.
\end{proof}

\begin{remark}[Minimax optimality]
\label{rem:optimality}
\memsad{}'s calibration (Theorem~\ref{thm:calibration_bound}) achieves reliable detection with $N = O(\log(1/\delta)/\rho^2)$ samples; the lower bound requires $N = \Omega(1/\rho^2)$. Thus \memsad{} is minimax-optimal up to $\log(1/\delta)$.
\end{remark}

\subsection{Calibration bounds and online regret}

\begin{theorem}[Calibration bound]
\label{thm:calibration_bound}
For $N$ i.i.d.\ benign scores with mean $\mu$, variance $\sigma^2$, and threshold $\tau_N = \hat{\mu}_N + \kappa\hat{\sigma}_N$, with probability $\geq 1 - \delta$:
\begin{equation}
  |\tau_N - \tau^*| \leq \sigma \left(\frac{1}{\sqrt{N}} + \frac{\kappa}{\sqrt{2(N-1)}}\right) \sqrt{2 \log(4/\delta)},
  \label{eq:calibration_bound}
\end{equation}
where $\tau^* = \mu + \kappa\sigma$. Proof via Hoeffding + chi-squared concentration with union bound (Appendix~\ref{app:proof_calibration}).
\end{theorem}

\begin{theorem}[Online calibration regret]
\label{thm:regret}
Let queries arrive online with per-step drift $\|\EE[\enc(q_t)] - \EE[\enc(q_{t+1})]\| \leq \Delta$. The cumulative regret of \memsad{}'s rolling window satisfies:
\begin{equation}
  \Regret(T) = \sum_{t=1}^{T} |\hat{\tau}_t - \tau_t^*|
  \leq T \cdot O\!\left(\frac{\sigma}{\sqrt{m_{\max}}} + m_{\max} \cdot \Delta \right).
  \label{eq:regret}
\end{equation}
The optimal window $m_{\max}^* = \Theta((\sigma/\Delta)^{2/3})$ minimizes per-step regret to $O(\sigma^{2/3}\Delta^{1/3})$. Proof in Appendix~\ref{app:proof_regret}.

\begin{corollary}[Non-asymptotic convergence]
\label{cor:convergence}
After $T$ queries with optimal window $m_{\max}^*$, the average calibration error satisfies:
\begin{equation}
  \frac{1}{T}\sum_{t=1}^{T} |\hat{\tau}_t - \tau_t^*| \leq C \sigma^{2/3}\Delta^{1/3} + \frac{\sigma\sqrt{2\log(4T/\delta)}}{\sqrt{m_{\max}^*}},
  \label{eq:convergence}
\end{equation}
with probability $\geq 1 - \delta$, where $C$ is a universal constant. The second term decays as $O(T^{-1/3}\sqrt{\log T})$ for $\delta = 1/T$.
\end{corollary}
\end{theorem}

\subsection{Generalization to unseen attacks}

The preceding results hold for a fixed adversary. We now bound \memsad{}'s detection guarantee for \emph{unseen} attack distributions.

\begin{theorem}[Generalization bound]
\label{thm:generalization}
Let $\cP_{\text{train}} = \{P_1, \ldots, P_K\}$ be $K$ attack distributions observed during calibration, and $P_{\text{new}}$ an unseen attack with $W_1(P_{\text{new}}, P_j) \leq \epsilon_W$ for some $P_j \in \cP_{\text{train}}$, where $W_1$ denotes the Wasserstein-1 distance on score distributions. Then:
\begin{equation}
  |\tpr_{\text{new}} - \tpr_j| \leq \frac{\epsilon_W}{\hat{\sigma}} + O\!\left(\frac{1}{\sqrt{N}}\right),
  \label{eq:generalization}
\end{equation}
where $\tpr_j$ is \memsad{}'s TPR on $P_j$ and $N$ is the calibration size.
\end{theorem}

\begin{proof}
The detection function $\pi(c) = \mathbf{1}[s(c; \cH) > \tau]$ is 1-Lipschitz in $s$ (indicator threshold). By Kantorovich-Rubinstein duality, $|\EE_{P_{\text{new}}}[\pi] - \EE_{P_j}[\pi]| \leq W_1(P_{\text{new}}, P_j)/\hat{\sigma}$ (normalizing by the calibration scale). The $O(1/\sqrt{N})$ term accounts for finite-sample threshold estimation (Theorem~\ref{thm:calibration_bound}).
\end{proof}

This guarantees that if a new attack's score distribution is $\epsilon_W$-close (in Wasserstein distance) to any previously observed attack, \memsad{}'s TPR transfers with bounded degradation. In our experiments, \minja{} and \injecmem{} have $W_1 \approx 0.03\hat{\sigma}$, predicting $|\Delta\tpr| \leq 0.03$, consistent with the observed TPR stability across attacks with similar access models.

\subsection{The discrete synonym loophole}

\begin{proposition}[Synonym-invariance loophole]
\label{prop:synonym}
Let $\enc$ satisfy $\|\enc(\mathrm{sub}(c, w, w')) - \enc(c)\| \leq \epsilon_{\text{syn}}$ for synonym substitutions. For $r$ substitutions yielding $c'$:
\begin{equation}
  |\cR(c') - \cR(c)| \leq r \epsilon_{\text{syn}}, \qquad
  |s(c'; \cH) - s(c; \cH)| \leq r \epsilon_{\text{syn}}.
  \label{eq:synonym_bounds}
\end{equation}
When $\epsilon_{\text{syn}} \approx 0$, the adversary evades detection while preserving retrieval via non-gradient lexical variation.
\end{proposition}

\begin{proof}
Cauchy-Schwarz gives $|\enc(q)^\top(\enc(c') - \enc(c))| \leq \|\enc(c') - \enc(c)\|$, and triangle inequality over $r$ substitutions yields $\|\enc(c') - \enc(c)\| \leq r\epsilon_{\text{syn}}$.
\end{proof}

This does \emph{not} contradict Theorem~\ref{thm:gradient_coupling}: the coupling holds for \emph{continuous} perturbations in $\RR^d$, while synonym substitution operates in the \emph{discrete} token space where $\epsilon_{\text{syn}} \approx 0$ for paraphrase-trained encoders.

\section{Discussion and limitations}
\label{sec:discussion}

\paragraph{No single defense dominates.}
The $3 \times 5$ matrix reveals substantial variation: watermark catches unsigned writes but fails if the attacker bypasses ingestion; \memsad{} excels against \minja{} (AUROC $= 1.000$) but requires triggered calibration for \agentpoison{}; proactive detection is complementary.
Composite portfolios cover all gaps, achieving $\asrr^* = 0$ (Definition~\ref{def:stackelberg}).

\paragraph{Certified detection and minimax optimality.}
Lemma~\ref{lem:certified} provides a checkable certified detection condition; Theorem~\ref{thm:minimax_lower} shows \emph{any} threshold detector requires $\Omega(1/\rho^2)$ samples, matched by \memsad{} up to $\log(1/\delta)$ (Remark~\ref{rem:optimality}).
The practical implication: maximize the SNR $\rho$ (e.g., triggered calibration raises $\rho$ from $\sim 1.2$ to $\sim 3.8$ for \agentpoison{}) rather than designing more complex algorithms.

\paragraph{Online adaptivity and deployment.}
Theorem~\ref{thm:regret} yields $m_{\max}^* \approx 46$ queries ($\Delta \approx 10^{-3}$, $\sigma \approx 0.1$), per-step regret $\approx 0.02$.
Triggered calibration shifts the threshold ($0.482 \to 0.432$), recovering $\tpr = 1.00$.
Recommended stack: (i)~watermark, (ii)~rolling \memsad{}, (iii)~composite ensemble ($\sim 200$ms overhead, Appendix~\ref{app:complexity}).

\paragraph{Limitations.}
(i)~200-entry corpus; Appendix~\ref{app:ablation} validates $|\cM| \in \{50, \ldots, 500\}$.
(ii)~$\asra$ modelled; local GPT-2 evaluator provides lower bounds.
(iii)~Slowly drifting assumption (Theorem~\ref{thm:regret}); abrupt shifts degrade calibration.
(iv)~Synonym loophole (Proposition~\ref{prop:synonym}) motivates joint semantic-lexical detection.
(v)~Main experiments use MiniLM-L6; Appendix~\ref{app:encoder_gen} validates 7 encoders.

\section{Proofs of main results}
\label{app:proofs}

\subsection{Proof of Theorem~\ref{thm:gradient_coupling} (geometric interpretation)}
\label{app:proof_gradient}

Under L2-normalization, embeddings lie on $\mathbb{S}^{d-1}$. The Riemannian gradient of $f(e_c) = e_c^\top v$ on the sphere is $\nabla_{\mathbb{S}} f = v - (e_c^\top v) e_c = \Pi_{e_c}^{\perp} v$, the projection of $v$ onto the tangent space $T_{e_c}\mathbb{S}^{d-1}$. Since detection and retrieval share the same inner product with $v = \enc(q^*)$, gradient coupling is a consequence of the shared geometry on the unit sphere.

\subsection{Proof of Theorem~\ref{thm:minimax_lower} (detailed)}
\label{app:proof_minimax}

\paragraph{Setup.} Instantiate Le Cam's method with:
$H_0$: $s(c; \cH) \sim P_0 = \cN(\mu_0, \sigma^2)$ (benign);
$H_1$: $s(c; \cH) \sim P_1 = \cN(\mu_1, \sigma^2)$ (poisoned), $\mu_1 > \mu_0$.
Then $\KL(P_0 \| P_1) = (\mu_1 - \mu_0)^2 / (2\sigma^2) = \rho^2/2$.

\paragraph{Application.} For product distributions, $\KL(P_0^N \| P_1^N) = N\rho^2/2$. By Pinsker: $\TV(P_0^N, P_1^N) \leq \sqrt{N\rho^2/4} = \rho\sqrt{N}/2$. Le Cam gives minimax risk $R^* \geq \frac{1}{2}(1 - \TV) \geq \frac{1}{2}(1 - \rho\sqrt{N}/2)$. Total error $\geq 2R^* = 1 - \rho\sqrt{N}/2$.

\paragraph{Alternative via Fano.} For $M = 2$ hypotheses, Fano's inequality gives $P_e \geq 1 - (I(X; \Theta) + \log 2)/\log M = 1 - (N\rho^2/4 + \log 2)/\log 2$, where $I(X; \Theta) \leq \max_j \KL(P_j^N \| \bar{P}^N) \leq N\rho^2/4$ by the mixture bound. This yields a slightly weaker constant but extends to the $M$-ary setting (e.g., distinguishing $M$ attack types simultaneously).

\paragraph{Tightness.} For Gaussian $P_0, P_1$, the Neyman-Pearson optimal test achieves $\fpr + (1-\tpr) = 2\Phi(-\rho\sqrt{N}/2)$ where $\Phi$ is the Gaussian CDF. Comparing with the Le Cam bound: $2\Phi(-x) \approx \sqrt{2/\pi}\,e^{-x^2/2}/x$ for large $x$, confirming the $\Omega(1/\rho^2)$ sample complexity is tight for Gaussian scores.

\subsection{Proof of Theorem~\ref{thm:calibration_bound}}
\label{app:proof_calibration}

By Hoeffding on $[0,1]$: $|\hat{\mu}_N - \mu| \leq \sqrt{\log(2/\delta)/(2N)}$ w.p.\ $\geq 1 - \delta/2$. By chi-squared concentration: $|\hat{\sigma}_N - \sigma| \leq \sigma\sqrt{\log(2/\delta)/(2(N-1))}$ w.p.\ $\geq 1 - \delta/2$. Triangle inequality: $|\tau_N - \tau^*| \leq |\hat{\mu}_N - \mu| + \kappa|\hat{\sigma}_N - \sigma|$. Union bound yields Eq.~\eqref{eq:calibration_bound}.

\subsection{Proof of Theorem~\ref{thm:regret}}
\label{app:proof_regret}

Decompose: $|\hat{\tau}_t - \tau_t^*| \leq \underbrace{|\hat{\tau}_t - \tau_t^{\cH}|}_{\text{estimation}} + \underbrace{|\tau_t^{\cH} - \tau_t^*|}_{\text{drift}}$.
Estimation error: $O(\sigma(1+\kappa)/\sqrt{m_{\max}})$ by Theorem~\ref{thm:calibration_bound}.
Drift error: cosine similarity is 1-Lipschitz on $\mathbb{S}^{d-1}$, so $|\mu_t^{\cH} - \mu_t| \leq m_{\max}\Delta$.
Setting equal: $\sigma/\sqrt{m_{\max}} = m_{\max}\Delta$ gives $m_{\max}^* = (\sigma/\Delta)^{2/3}$, yielding per-step regret $O(\sigma^{2/3}\Delta^{1/3})$.

\subsection{Detection-evasion tradeoff surface}
\label{app:tradeoff_surface}

The tradeoff in Corollary~\ref{cor:tradeoff} defines a surface in $(\epsilon, \theta, \Delta\cD)$ space where $\theta$ is the angle between perturbation $\delta$ and gradient $\nabla\cD$. Expanding to second order:
\begin{equation}
  \cD(e_c + \epsilon\delta) - \cD(e_c) = \epsilon\|\nabla\cD\|\cos\theta + \frac{\epsilon^2}{2}\delta^\top H_{\cD}\,\delta + O(\epsilon^3),
\end{equation}
where $H_{\cD}$ is the Hessian of the cosine similarity on the sphere. Under L2-normalization, $H_{\cD} = -(e_c^\top \enc(q^*))(I - e_c e_c^\top)/\|e_c\|^2$, giving eigenvalues $\lambda_i = -(e_c^\top \enc(q^*))$ for directions orthogonal to $e_c$ (and $0$ in the radial direction). The evasion region $\cD(e_c') < \cD(e_c)$ requires $\cos\theta < -\epsilon\|H_{\cD}\|/(2\|\nabla\cD\|)$, which contracts as $\epsilon \to 0$, confirming the infinitesimal impossibility of evasion without retrieval degradation.

\subsection{Proof of Theorem~\ref{thm:generalization} (detailed)}
\label{app:proof_generalization}

\paragraph{Setup.} Let $F_j$ and $F_{\text{new}}$ denote the CDFs of scores $s(c; \cH)$ under $P_j$ and $P_{\text{new}}$, respectively. The TPR at threshold $\tau$ is $\tpr(\tau) = 1 - F(\tau)$.

\paragraph{Wasserstein coupling.} By Kantorovich-Rubinstein duality:
\begin{equation}
  W_1(F_j, F_{\text{new}}) = \sup_{\|f\|_{\text{Lip}} \leq 1} |\EE_{F_j}[f] - \EE_{F_{\text{new}}}[f]| = \int_{-\infty}^{\infty} |F_j(s) - F_{\text{new}}(s)| \, ds.
\end{equation}
Since $\tpr_j = 1 - F_j(\tau)$ and $\tpr_{\text{new}} = 1 - F_{\text{new}}(\tau)$:
$|\tpr_j - \tpr_{\text{new}}| = |F_{\text{new}}(\tau) - F_j(\tau)| \leq W_1(F_j, F_{\text{new}})/\hat{\sigma}$,
where the $1/\hat{\sigma}$ normalization arises because the detection operates on the standardized score $(s - \hat{\mu})/\hat{\sigma}$ and the Wasserstein distance scales inversely with the standard deviation of the reference distribution.

\paragraph{Threshold estimation.} By Theorem~\ref{thm:calibration_bound}, $|\hat{\tau} - \tau^*| \leq \eta_N$ w.p.\ $\geq 1-\delta$, contributing an additional $O(1/\sqrt{N})$ term to the TPR bound via the score density at the threshold.

\paragraph{Corollary~\ref{cor:convergence}.} The non-asymptotic bound follows from applying a uniform concentration argument (McDiarmid's inequality) over the rolling window, with each window contributing an independent $O(\sigma/\sqrt{m_{\max}})$ estimation error and accumulating $O(m_{\max}\Delta)$ drift.

\subsection{Stackelberg equilibrium existence}
\label{app:equilibrium}

\begin{proposition}[Equilibrium existence]
\label{prop:equilibrium}
The Stackelberg game $\cG$ (Definition~\ref{def:stackelberg}) admits a subgame-perfect equilibrium $(\pi_{\cD}^*, \cP^*)$ when $\Pi$ is restricted to threshold detectors with $\kappa \in [\kappa_{\min}, \kappa_{\max}]$.
\end{proposition}

\begin{proof}
The defender's strategy space $\Pi_\kappa = \{\pi_\kappa : \kappa \in [\kappa_{\min}, \kappa_{\max}]\}$ is compact. For fixed $\pi_\kappa$, the adversary's best-response set $\cB(\pi_\kappa) = \argmax_{\cP} \asrr(\cP \setminus \{p : s(p; \cH) > \hat{\mu} + \kappa\hat{\sigma}\}, \cQ_v)$ is nonempty (the empty set is always feasible) and upper hemicontinuous in $\kappa$ since $s(p; \cH)$ is continuous and the threshold $\hat{\mu} + \kappa\hat{\sigma}$ is linear in $\kappa$. The defender's objective $\max_{\cP \in \cB(\pi_\kappa)} \asrr(\cP, \cQ_v)$ is upper semicontinuous by the Berge maximum theorem. By the extreme value theorem, $\pi_{\cD}^* = \argmin_{\kappa \in [\kappa_{\min}, \kappa_{\max}]} \max_{\cP \in \cB(\pi_\kappa)} \asrr$ exists.
\end{proof}

In practice, the composite portfolio $(\pi_{\text{wm}} \lor \pi_{\text{SAD}} \lor \pi_{\text{pro}})$ achieves $\asrr^* = 0$ against all three attacks (Table~\ref{tab:defense_tpr_fpr}), representing the defender's ideal equilibrium where the adversary's best response yields zero utility.

\subsection{Finite-sample FPR concentration}
\label{app:fpr_concentration}

\begin{proposition}[FPR concentration]
\label{prop:fpr_concentration}
Let $c_1, \ldots, c_n$ be $n$ i.i.d.\ benign candidates with scores $s_i = s(c_i; \cH)$. Let $\hat{\text{FPR}}_n = n^{-1}\sum_{i=1}^n \mathbf{1}[s_i > \hat{\mu} + \kappa\hat{\sigma}]$. Under sub-Gaussian scores with parameter $\sigma^2$:
\begin{equation}
  \PP\!\left[|\hat{\text{FPR}}_n - \text{FPR}| > \epsilon\right] \leq 2\exp\!\left(-\frac{n\epsilon^2}{2\text{FPR}(1-\text{FPR})}\right) + 2\exp\!\left(-\frac{n\epsilon^2}{8}\right).
\end{equation}
The first term is the Bernstein bound on the empirical FPR; the second accounts for threshold estimation error (from Theorem~\ref{thm:calibration_bound}). For $\kappa = 2.0$ and $n = 1000$: $\PP[|\hat{\text{FPR}} - 0| > 0.01] < 0.006$, consistent with the Clopper-Pearson CI $[0.000, 0.004]$ (Table~\ref{tab:fpr_validation}).
\end{proposition}

\section{Graph memory attacks}
\label{app:graph}

\begin{definition}[Graph memory system]
A \emph{graph memory} is $(\cG, \cM, \enc, k)$ where $\cG = (V, E, \phi)$ is a knowledge graph with attribute function $\phi : V \cup E \to \cV^*$. Retrieval combines embedding similarity with BFS traversal:
$\Ret_{\cG}(q, k, h) = \{v \in V : d_{\cG}(v, v^*) \leq h,\; v^* \in \argmax_{v'} \cosim(\enc(q), \enc(\phi(v')))\}$.
\end{definition}

Three structural attacks: \textbf{Hub insertion} (high-degree adversarial node), \textbf{edge hijack} (shortcuts to adversarial nodes), \textbf{subgraph cluster} (self-reinforcing adversarial community). Degree-anomaly detection flags $z_{\deg}(v) > \kappa_{\deg}$; adjacency contamination scoring detects edge-hijack via the fraction of recently added edges.

\section{Multi-agent propagation}
\label{app:propagation}

\begin{definition}[SIR propagation]
For $N$ agents sharing $\cM_{\text{shared}}$, the discrete-step SIR model~\citep{gu2024agentsmith} evolves: $S(t{+}1) = S(t) - \beta S(t) I(t)/N$, $I(t{+}1) = I(t) + \beta S(t) I(t)/N - \gamma I(t)$, $R(t{+}1) = R(t) + \gamma I(t)$, where $\beta$ is transmission rate and $\gamma$ recovery rate.
\end{definition}

Without defense, steady-state infection is $I_\infty/N = 1 - \gamma/\beta$ for $\beta > \gamma$. With \memsad{} quarantine ($\tpr = 1.00$ for \minja{}), $\gamma_{\text{eff}} = \gamma + \tpr \cdot \beta$; the condition $\beta > \gamma_{\text{eff}}$ is never satisfied, eliminating secondary infection.
